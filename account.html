function initEditModal() { // 新增：编辑弹窗图片处理 let editUploadedImage = null; const editExpenseImage = document.getElementById('editExpenseImage'); const editImageUploadArea = document.getElementById('editImageUploadArea'); const editPreviewImageContainer = document.getElementById('editPreviewImageContainer'); const editPreviewImage = document.getElementById('editPreviewImage'); const editRemoveImage = document.getElementById('editRemoveImage'); // 编辑弹窗图片上传点击 editImageUploadArea.addEventListener('click', () => { editExpenseImage.click(); }); // 编辑弹窗图片预览 editExpenseImage.addEventListener('change', (e) => { const file = e.target.files[0]; if (file) { const maxSize = 5 * 1024 * 1024; if (file.size > maxSize) { showToast('图片大小不能超过5MB', 'error'); return; } const reader = new FileReader(); reader.onload = (event) => { editUploadedImage = event.target.result; editPreviewImage.src = editUploadedImage; editPreviewImageContainer.classList.remove('hidden'); }; reader.readAsDataURL(file); } }); // 编辑弹窗移除图片 editRemoveImage.addEventListener('click', () => { editUploadedImage = null; editExpenseImage.value = ''; editPreviewImageContainer.classList.add('hidden'); }); // 原有编辑弹窗代码... // 修改编辑表单提交逻辑，添加图片更新 editForm.addEventListener('submit', function(e) { e.preventDefault(); const id = document.getElementById('edit-id').value; const payer = document.getElementById('edit-payer').value; const description = document.getElementById('edit-description').value.trim(); const amount = parseFloat(document.getElementById('edit-amount').value); const date = document.getElementById('edit-date').value; if (!description) { showToast('请输入费用描述', 'error'); return; } if (!amount || amount <= 0) { showToast('请输入有效的金额', 'error'); return; } // 更新费用 const index = expenses.findIndex(expense => expense.id === id); if (index !== -1) { expenses[index].payer = payer; expenses[index].description = description; expenses[index].amount = amount; expenses[index].date = date; // 更新图片（如果有新上传的图片） if (editUploadedImage) { expenses[index].imageUrl = editUploadedImage; } saveExpenses(); updateSummary(); updateAnalysis(); editModal.classList.add('hidden'); showToast('费用已更新'); // 重置编辑图片状态 editUploadedImage = null; editExpenseImage.value = ''; } }); // 编辑按钮点击时加载图片 document.addEventListener('click', function(e) { if (e.target.closest('.edit-expense')) { const id = e.target.closest('.edit-expense').dataset.id; const expense = expenses.find(exp => exp.id === id); if (expense) { // 原有赋值代码... document.getElementById('edit-id').value = expense.id; document.getElementById('edit-category').value = expense.category; document.getElementById('edit-payer').value = expense.payer; document.getElementById('edit-description').value = expense.description; document.getElementById('edit-amount').value = expense.amount; document.getElementById('edit-date').value = expense.date; // 加载图片预览 if (expense.imageUrl) { editPreviewImage.src = expense.imageUrl; editPreviewImageContainer.classList.remove('hidden'); editUploadedImage = expense.imageUrl; } else { editPreviewImageContainer.classList.add('hidden'); editUploadedImage = null; } editModal.classList.remove('hidden'); } } }); // 原有删除按钮代码... }
